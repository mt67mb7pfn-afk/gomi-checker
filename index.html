<script>

// ============================================================

//  設定：GASデプロイ後のURLをここに貼る

// ============================================================

const API_URL = 'https://script.google.com/macros/s/AKfycbyVJ_yM_arZtDVgEts-ufmwMsZHmlKXOL3ITkg9NTXia1OsRXznm7s-nWJdiCaf-PYc/exec';


// ============================================================

//  定数

// ============================================================

const DAYS_JA = ['日','月','火','水','木','金','土'];

const WEEK_DAYS = ['月','火','水','木','金','土','日'];


const GOMI_TYPES = [

  {

    key: 'moyas',

    label: '燃やすごみ',

    sub: '燃えないごみ・スプレー缶・乾電池も同日',

    cssClass: 'moyas',

    getDays: a => a.moyas_days

  },

  {

    key: 'plastic',

    label: 'プラスチック資源',

    sub: '容器包装 + プラスチック製品（2025年4月〜新ルール）',

    cssClass: 'plastic',

    getDays: a => a.plastic_day ? [a.plastic_day] : []

  },

  {

    key: 'can',

    label: '缶・びん・ペットボトル',

    sub: '小さな金属類も同日',

    cssClass: 'can',

    getDays: a => a.can_day ? [a.can_day] : []

  }

];


// ============================================================

//  状態

// ============================================================

let allAreas = [];

let selectedAreaId = localStorage.getItem('selectedArea') || '';


// ============================================================

//  初期化

// ============================================================

async function init() {

  try {

    const res = await fetch(API_URL);

    const data = await res.json();

    allAreas = data.areas || [];


    // 前回選択を復元、なければ先頭

    if (!allAreas.find(a => a.id === selectedAreaId)) {

      selectedAreaId = allAreas[0]?.id || '';

    }


    render();

  } catch (e) {

    document.getElementById('app').innerHTML =

      '<div class="error">データの読み込みに失敗しました。<br>しばらく経ってから再度お試しください。</div>';

  }

}


// ============================================================

//  レンダリング

// ============================================================

function render() {

  const app = document.getElementById('app');

  const today = new Date();

  const todayDow = DAYS_JA[today.getDay()]; // "火" など

  const area = allAreas.find(a => a.id === selectedAreaId);


  // 日付バナー

  const dateStr = `${today.getMonth()+1}月${today.getDate()}日（${todayDow}）`;

  const dateHtml = `

    <div class="date-banner">

      <div class="date-today">${dateStr}</div>

      <div class="date-sub">今日の収集日情報</div>

    </div>`;


  // 自治会セレクター

  const options = allAreas.map(a =>

    `<option value="${a.id}" ${a.id === selectedAreaId ? 'selected' : ''}>${a.name}</option>`

  ).join('');

  const selectorHtml = `

    <div class="selector-wrap">

      <div class="selector-label">自治会を選択</div>

      <select onchange="changeArea(this.value)">${options}</select>

    </div>`;


  if (!area) {

    app.innerHTML = dateHtml + selectorHtml + '<div class="error">自治会データが見つかりません</div>';

    return;

  }


  // 収集カード

  const cardsHtml = GOMI_TYPES.map(type => {

    const days = type.getDays(area);

    const isToday = days.includes(todayDow);

    const dayLabel = days.length ? days.join('・') + '曜日' : '—';


    return `

      <div class="card ${isToday ? 'card-today ' + type.cssClass : 'card-none'}">

        <div class="card-head">

          <div class="card-type">${type.label}</div>

          <div class="card-status">${isToday ? '今日収集' : '収集なし'}</div>

        </div>

        <div class="card-detail">

          収集日：${dayLabel}

          ${type.sub ? `<br>${type.sub}` : ''}

          ${area.note && type.key === 'moyas' ? `<br>備考：${area.note}` : ''}

        </div>

      </div>`;

  }).join('');


  // 週間ドットカレンダー

  const weekHtml = buildWeekCalendar(area, todayDow);


  // 凡例

  const legendHtml = `

    <div class="legend">

      <div class="legend-item"><div class="legend-dot" style="background:var(--c-moyas)"></div>燃やすごみ</div>

      <div class="legend-item"><div class="legend-dot" style="background:var(--c-plastic)"></div>プラスチック資源</div>

      <div class="legend-item"><div class="legend-dot" style="background:var(--c-can)"></div>缶・びん・ペット</div>

    </div>`;


  // フッター

  const footerHtml = `

    <div class="footer">

      朝8時までに集積場所へ出してください<br>

      <a href="https://www.city.yokohama.lg.jp/kurashi/sumai-kurashi/gomi-recycle/gomi/shushuyobi/tsuzuki/" target="_blank">

        横浜市公式：都筑区の収集曜日

      </a>

    </div>`;


  app.innerHTML = dateHtml + selectorHtml +

    `<div class="cards">${cardsHtml}</div>` +

    weekHtml + legendHtml + footerHtml;

}


// 週間ドットカレンダーを生成

function buildWeekCalendar(area, todayDow) {

  const cols = WEEK_DAYS.map(dow => {

    const dots = [];

    if (area.moyas_days.includes(dow))     dots.push('<div class="dot dot-moyas"></div>');

    if (area.plastic_day === dow)          dots.push('<div class="dot dot-plastic"></div>');

    if (area.can_day === dow)              dots.push('<div class="dot dot-can"></div>');

    if (dots.length === 0)                 dots.push('<div class="dot dot-empty"></div>');


    const isToday = dow === todayDow;

    return `

      <div class="week-day">

        <div class="week-day-label ${isToday ? 'today-label' : ''}">${dow}</div>

        <div class="week-day-dots">${dots.join('')}</div>

      </div>`;

  }).join('');


  return `

    <div class="week-section">

      <div class="week-title">今週の収集スケジュール</div>

      <div class="week-row">${cols}</div>

    </div>`;

}


// 自治会変更

function changeArea(id) {

  selectedAreaId = id;

  localStorage.setItem('selectedArea', id);

  render();

}


// 起動

init();
